from PySide2.QtWidgets import QMainWindow
from PySide2 import QtWidgets

import sys

if __name__ == '__main__':
    app = QtWidgets.QApplication(sys.argv)
    window = QMainWindow()
    window.setWindowTitle('Pyside2 with Pyinstaller')
    window.resize(250, 150)
    window.show()
    sys.exit(app.exec_())
